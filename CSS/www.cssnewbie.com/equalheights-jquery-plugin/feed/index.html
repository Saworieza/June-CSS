]]> By: Chris Carvache http://www.cssnewbie.com/equalheights-jquery-plugin/comment-page-3/#comment-15415 Thu, 18 Jul 2013 05:19:44 +0000 http://www.cssnewbie.com/?p=292#comment-15415 Here’s a detailed example on how to make this plugin responsive:


/**
* Multiple Column Nesting / Respnosive Example
* ==============================================
* This example demonstates the
* following techniques:
*
* 1. How to nest columns
* 2. How to make them responsive
* 3. How to conditionally use the plugin
* using Modernizr
*
* NOTE: This is definitely not the most effecient
* way to do this from a speed perspective but, its
* functional and concise.
*/

jQuery(document).ready(function($){

// Our function for properly handling calculating columns
function calculateColumns() {

// We'll use modernizr to make our resizing logic responsive
// Use your breakpoints appropriately please
if (Modernizr.mq('(min-width: 768px)') ) {

// the third columns are nested within the second columns
// they must be called FIRST so that the second columns
// (the parent columns) can correctly calculate the height
$(".first-columns").removeAttr('style').equalHeights();
$(".third-columns").removeAttr('style').equalHeights();
$(".second-columns").removeAttr('style').equalHeights();

} else {

// Twitter Bootstrap usually uses a one column
// layout 767px and below, so we'll just make sure
// the columns don't have any inappropriate styling
$(".first-columns").removeAttr('style');
$(".third-columns").removeAttr('style');
$(".second-columns").removeAttr('style');

}
}

// Let's bind and handle our resizing
$(window).resize(function(){
calculateColumns();
});

// This sets up all of our columns on load
calculateColumns();

});
